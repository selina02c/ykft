Backtracking

1. Problema Permutărilor

( Găsiți toate permutările unui șir dat.)

#include (iostream)

#include (vector)

void permutari(std::vector& sir, int index) {

if (index == sir.size() - 1) {

for (int numar : sir) {

std::cout << numar << " ";

}

std::cout << std::endl;

return;

}

for (int i = index; i < sir.size(); ++i) {

std::swap(sir[index], sir[i]);

permutari(sir, index + 1);

std::swap(sir[index], sir[i]);

}

}

int main() {

std::vector sir = {1, 2, 3};

std::cout << "Permutarile sirului sunt:\n";

permutari(sir, 0);

return 0;

}

3. Problema Reuniunii a Două Șiruri

( Găsiți toate modurile de a forma un șir folosind intercalarea a două șiruri date.)

#include (iostream)

#include (vector)

#include (string)

void backtrackingInterclasare(std::string& rezultat, const std::string& s1, const std::string& s2, int i, int j) {

if (i == s1.length() && j == s2.length()) {

std::cout << rezultat << std::endl;

return;

}

if (i < s1.length()) {

rezultat.push_back(s1[i]);

backtrackingInterclasare(rezultat, s1, s2, i + 1, j);

rezultat.pop_back();

}

if (j < s2.length()) {

rezultat.push_back(s2[j]);

backtrackingInterclasare(rezultat, s1, s2, i, j + 1);

rezultat.pop_back();

}

}

void interclasareSiruri(const std::string& s1, const std::string& s2) {

std::string rezultat;

backtrackingInterclasare(rezultat, s1, s2, 0, 0);

}

int main() {

std::string s1 = "abc";

std::string s2 = "def";

std::cout << "Rezultatele interclasarii sunt:\n";

interclasareSiruri(s1, s2);

return 0;

}

2. Problema Ghiozdanului Fracționat

( Alegeți obiecte dintr-o listă astfel încât să umpleți un ghiozdan până la o capacitate dată maximă, obținând valoarea maximă, având opțiunea de a lua fracțiuni din obiecte.)

#include (iostream)

#include (vector)

#include (algorithm)

struct Obiect {

int greutate;

int valoare;

};

bool comparaRaport(Obiect a, Obiect b) {

double raportA = (double)a.valoare / a.greutate;

double raportB = (double)b.valoare / b.greutate;

return raportA > raportB;

}

double ghiozdanFactionat(std::vector& obiecte, int capacitate) {

std::sort(obiecte.begin(), obiecte.end(), comparaRaport);

double valoareTotala = 0.0;

for (const auto& obiect : obiecte) {

if (capacitate >= obiect.greutate) {

capacitate -= obiect.greutate;

valoareTotala += obiect.valoare;

} else {

double fractie = (double)capacitate / obiect.greutate;

valoareTotala += obiect.valoare * fractie;

break;

}

}

return valoareTotala;

}

int main() {

std::vector obiecte = {{10, 60}, {20, 100}, {30, 120}};

int capacitateGhiozdan = 50;

double valoareMaxima = ghiozdanFactionat(obiecte,

capacitateGhiozdan);

std::cout << "Valoare maxima ce poate fi pusa in ghiozdan: "

<< valoareMaxima << std::endl;

return 0;

}

3. Problema Programării Activităților

(Alegeți un set de activități care să nu se suprapună, maximizând numărul de activități selectate.)

#include (iostream)

#include (vector)

#include (algorithm)

struct Activitate {

int inceput;

int sfarsit;

};

bool comparaSfarsit(Activitate a, Activitate b) {

return a.sfarsit < b.sfarsit;

}

int numarMaximActivitati(std::vector& activitati) {

if (activitati.empty()) return 0;

std::sort(activitati.begin(), activitati.end(), comparaSfarsit);

int numarActivitati = 1;

int activitateSelectata = 0;

for (int i = 1; i < activitati.size(); ++i) {

if (activitati[i].inceput >= activitati[activitateSelectata].sfarsit) { ++numarActivitati;

activitateSelectata = i;

}

}

return numarActivitati;

}

int main() {

std::vector activitati = {{1, 4}, {3, 5}, {0, 6}, {5, 7}, {3, 8}, {5, 9}, {6, 10}, {8, 11}, {8, 12}, {2, 13}, {12, 14}};

std::cout << "Numarul maxim de activitati selectate: "

<< numarMaximActivitati(activitati) << std::endl;

return 0;

}
